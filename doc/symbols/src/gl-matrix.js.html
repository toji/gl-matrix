227 /**
228  * Generates a unit vector of the same direction as the provided vec3
229  * If vector length is 0, returns [0, 0, 0]
230  *
231  * @param {vec3} vec vec3 to normalize
232  * @param {vec3} [dest] vec3 receiving operation result. If not specified result is written to vec
233  *
234  * @returns {vec3} dest if specified, vec otherwise
235  */
236 vec3.normalize = function (vec, dest) {
237     if (!dest) { dest = vec; }
238 
239     var x = vec[0], y = vec[1], z = vec[2],
240         len = Math.sqrt(x * x + y * y + z * z);
241 
242     if (!len) {
243         dest[0] = 0;
244         dest[1] = 0;
245         dest[2] = 0;
246         return dest;
247     } else if (len === 1) {
248         dest[0] = x;
249         dest[1] = y;
250         dest[2] = z;
251         return dest;
252     }
253 
254     len = 1 / len;
255     dest[0] = x * len;
256     dest[1] = y * len;
257     dest[2] = z * len;
258     return dest;
259 };

263  *

374 /**
375  * Projects the specified vec3 from screen space into object space
376  * Based on the <a href="http://webcvs.freedesktop.org/mesa/Mesa/src/glu/mesa/project.c?revision=1.4&view=markup">Mesa gluUnProject implementation</a>
377  *
378  * @param {vec3} vec Screen-space vector to project
379  * @param {mat4} view View matrix
380  * @param {mat4} proj Projection matrix
381  * @param {vec4} viewport Viewport as given to gl.viewport [x, y, width, height]
382  * @param {vec3} [dest] vec3 receiving unprojected result. If not specified result is written to vec
383  *
384  * @returns {vec3} dest if specified, vec otherwise
385  */
386 vec3.unproject = function (vec, view, proj, viewport, dest) {
387     if (!dest) { dest = vec; }
388 
389     var m = mat4.create();
390     var v = new MatrixArray(4);
391     
392     v[0] = (vec[0] - viewport[0]) * 2.0 / viewport[2] - 1.0;
393     v[1] = (vec[1] - viewport[1]) * 2.0 / viewport[3] - 1.0;
394     v[2] = 2.0 * vec[2] - 1.0;
395     v[3] = 1.0;
396     
397     mat4.multiply(proj, view, m);
398     if(!mat4.inverse(m)) { return null; }
399     
400     mat4.multiplyVec4(m, v);
401     if(v[3] === 0.0) { return null; }
402 
403     dest[0] = v[0] / v[3];
404     dest[1] = v[1] / v[3];
405     dest[2] = v[2] / v[3];
406     
407     return dest;
408 };

742  *
743  * @param {mat4} mat mat4 to calculate inverse of
744  * @param {mat4} [dest] mat4 receiving inverse matrix. If not specified result is written to mat
745  *
746  * @param {mat4} dest is specified, mat otherwise, null if matrix cannot be inverted
747  */
748 mat4.inverse = function (mat, dest) {
749     if (!dest) { dest = mat; }
750 
751     // Cache the matrix values (makes for huge speed increases!)
752     var a00 = mat[0], a01 = mat[1], a02 = mat[2], a03 = mat[3],
753         a10 = mat[4], a11 = mat[5], a12 = mat[6], a13 = mat[7],
754         a20 = mat[8], a21 = mat[9], a22 = mat[10], a23 = mat[11],
755         a30 = mat[12], a31 = mat[13], a32 = mat[14], a33 = mat[15],
756 
757         b00 = a00 * a11 - a01 * a10,

763         b06 = a20 * a31 - a21 * a30,

765         b08 = a20 * a33 - a23 * a30,
766         b09 = a21 * a32 - a22 * a31,
767         b10 = a21 * a33 - a23 * a31,
768         b11 = a22 * a33 - a23 * a32,
769 
770         d = (b00 * b11 - b01 * b10 + b02 * b09 + b03 * b08 - b04 * b07 + b05 * b06),
771         invDet;
772 
773         // Calculate the determinant
774         if (!d) { return null; }
775         invDet = 1 / d;

780     dest[3] = (-a21 * b05 + a22 * b04 - a23 * b03) * invDet;

782     dest[5] = (a00 * b11 - a02 * b08 + a03 * b07) * invDet;

786     dest[9] = (-a00 * b10 + a01 * b08 - a03 * b06) * invDet;

1065 /**
1066  * Rotates a matrix by the given angle around the specified axis
1067  * If rotating around a primary axis (X,Y,Z) one of the specialized rotation functions should be used instead for performance
1068  *
1069  * @param {mat4} mat mat4 to rotate
1070  * @param {number} angle Angle (in radians) to rotate
1071  * @param {vec3} axis vec3 representing the axis to rotate around 
1072  * @param {mat4} [dest] mat4 receiving operation result. If not specified result is written to mat
1073  *
1074  * @returns {mat4} dest if specified, mat otherwise
1075  */
1076 mat4.rotate = function (mat, angle, axis, dest) {

1078         len = Math.sqrt(x * x + y * y + z * z),
1079         s, c, t,
1080         a00, a01, a02, a03,
1081         a10, a11, a12, a13,
1082         a20, a21, a22, a23,
1083         b00, b01, b02,
1084         b10, b11, b12,
1085         b20, b21, b22;
1086 
1087     if (!len) { return null; }
1088     if (len !== 1) {
1089         len = 1 / len;
1090         x *= len;
1091         y *= len;
1092         z *= len;
1093     }
1094 
1095     s = Math.sin(angle);
1096     c = Math.cos(angle);
1097     t = 1 - c;
1098 
1099     a00 = mat[0]; a01 = mat[1]; a02 = mat[2]; a03 = mat[3];
1100     a10 = mat[4]; a11 = mat[5]; a12 = mat[6]; a13 = mat[7];
1101     a20 = mat[8]; a21 = mat[9]; a22 = mat[10]; a23 = mat[11];
1102

1104     b00 = x * x * t + c; b01 = y * x * t + z * s; b02 = z * x * t - y * s;

1107 
1108     if (!dest) {
1109         dest = mat;
1110     } else if (mat !== dest) { // If the source and destination differ, copy the unchanged last row
1111         dest[12] = mat[12];
1112         dest[13] = mat[13];
1113         dest[14] = mat[14];
1114         dest[15] = mat[15];
1115     }
1116 
1117     // Perform rotation-specific matrix multiplication
1118     dest[0] = a00 * b00 + a10 * b01 + a20 * b02;
1119     dest[1] = a01 * b00 + a11 * b01 + a21 * b02;
1120     dest[2] = a02 * b00 + a12 * b01 + a22 * b02;
1121     dest[3] = a03 * b00 + a13 * b01 + a23 * b02;
1122 
1123     dest[4] = a00 * b10 + a10 * b11 + a20 * b12;
1124     dest[5] = a01 * b10 + a11 * b11 + a21 * b12;
1125     dest[6] = a02 * b10 + a12 * b11 + a22 * b12;
1126     dest[7] = a03 * b10 + a13 * b11 + a23 * b12;
1127 
1128     dest[8] = a00 * b20 + a10 * b21 + a20 * b22;
1129     dest[9] = a01 * b20 + a11 * b21 + a21 * b22;
1130     dest[10] = a02 * b20 + a12 * b21 + a22 * b22;
1131     dest[11] = a03 * b20 + a13 * b21 + a23 * b22;

1133 };

1371 /**
1372  * Generates a look-at matrix with the given eye position, focal point, and up axis
1373  *
1374  * @param {vec3} eye Position of the viewer

1376  * @param {vec3} up vec3 pointing "up"
1377  * @param {mat4} [dest] mat4 frustum matrix will be written into
1378  *
1379  * @returns {mat4} dest if specified, a new mat4 otherwise
1380  */
1381 mat4.lookAt = function (eye, center, up, dest) {
1382     if (!dest) { dest = mat4.create(); }
1383 
1384     var x0, x1, x2, y0, y1, y2, z0, z1, z2, len,
1385         eyex = eye[0],
1386         eyey = eye[1],
1387         eyez = eye[2],
1388         upx = up[0],
1389         upy = up[1],
1390         upz = up[2],
1391         centerx = center[0],
1392         centery = center[1],
1393         centerz = center[2];
1394 
1395     if (eyex === centerx && eyey === centery && eyez === centerz) {
1396         return mat4.identity(dest);
1397     }
1398 
1399     //vec3.direction(eye, center, z);
1400     z0 = eyex - centerx;
1401     z1 = eyey - centery;
1402     z2 = eyez - centerz;
1403 
1404     // normalize (no check needed for 0 because of early return)
1405     len = 1 / Math.sqrt(z0 * z0 + z1 * z1 + z2 * z2);
1406     z0 *= len;
1407     z1 *= len;
1408     z2 *= len;
1409 
1410     //vec3.normalize(vec3.cross(up, z, x));
1411     x0 = upy * z2 - upz * z1;
1412     x1 = upz * z0 - upx * z2;
1413     x2 = upx * z1 - upy * z0;

1415     if (!len) {
1416         x0 = 0;
1417         x1 = 0;
1418         x2 = 0;
1419     } else {
1420         len = 1 / len;
1421         x0 *= len;
1422         x1 *= len;
1423         x2 *= len;
1424     }
1425 
1426     //vec3.normalize(vec3.cross(z, x, y));
1427     y0 = z1 * x2 - z2 * x1;
1428     y1 = z2 * x0 - z0 * x2;
1429     y2 = z0 * x1 - z1 * x0;
1430 
1431     len = Math.sqrt(y0 * y0 + y1 * y1 + y2 * y2);
1432     if (!len) {
1433         y0 = 0;
1434         y1 = 0;
1435         y2 = 0;
1436     } else {
1437         len = 1 / len;
1438         y0 *= len;
1439         y1 *= len;
1440         y2 *= len;
1441     }
1442 
1443     dest[0] = x0;
1444     dest[1] = y0;
1445     dest[2] = z0;
1446     dest[3] = 0;
1447     dest[4] = x1;
1448     dest[5] = y1;
1449     dest[6] = z1;
1450     dest[7] = 0;
1451     dest[8] = x2;
1452     dest[9] = y2;
1453     dest[10] = z2;
1454     dest[11] = 0;
1455     dest[12] = -(x0 * eyex + x1 * eyey + x2 * eyez);
1456     dest[13] = -(y0 * eyex + y1 * eyey + y2 * eyez);
1457     dest[14] = -(z0 * eyex + z1 * eyey + z2 * eyez);

1459 
1460     return dest;

1848 /**
1849  * Performs a spherical linear interpolation between two quat4
1850  *
1851  * @param {quat4} quat First quaternion
1852  * @param {quat4} quat2 Second quaternion
1853  * @param {number} slerp Interpolation amount between the two inputs
1854  * @param {quat4} [dest] quat4 receiving operation result. If not specified result is written to quat
1855  *
1856  * @returns {quat4} dest if specified, quat otherwise
1857  */
1858 quat4.slerp = function (quat, quat2, slerp, dest) {
1859     if (!dest) { dest = quat; }
1860 
1861     var cosHalfTheta = quat[0] * quat2[0] + quat[1] * quat2[1] + quat[2] * quat2[2] + quat[3] * quat2[3],
1862         halfTheta,
1863         sinHalfTheta,
1864         ratioA,
1865         ratioB;
1866 
1867     if (Math.abs(cosHalfTheta) >= 1.0) {
1868         if (dest !== quat) {
1869             dest[0] = quat[0];
1870             dest[1] = quat[1];
1871             dest[2] = quat[2];
1872             dest[3] = quat[3];
1873         }
1874         return dest;
1875     }
1876 
1877     halfTheta = Math.acos(cosHalfTheta);
1878     sinHalfTheta = Math.sqrt(1.0 - cosHalfTheta * cosHalfTheta);
1879 
1880     if (Math.abs(sinHalfTheta) < 0.001) {
1881         dest[0] = (quat[0] * 0.5 + quat2[0] * 0.5);
1882         dest[1] = (quat[1] * 0.5 + quat2[1] * 0.5);
1883         dest[2] = (quat[2] * 0.5 + quat2[2] * 0.5);
1884         dest[3] = (quat[3] * 0.5 + quat2[3] * 0.5);
1885         return dest;
1886     }
1887 
1888     ratioA = Math.sin((1 - slerp) * halfTheta) / sinHalfTheta;
1889     ratioB = Math.sin(slerp * halfTheta) / sinHalfTheta;
1890 
1891     dest[0] = (quat[0] * ratioA + quat2[0] * ratioB);
1892     dest[1] = (quat[1] * ratioA + quat2[1] * ratioB);
1893     dest[2] = (quat[2] * ratioA + quat2[2] * ratioB);
1894     dest[3] = (quat[3] * ratioA + quat2[3] * ratioB);
1895 
1896     return dest;
1897 };
