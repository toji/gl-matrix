 23 /**
 24  * @class 2x3 Matrix
 25  * @name mat2d
 26  * 
 27  * @description 
 28  * A mat2d contains six elements defined as:
 29  * <pre>
 30  * [a, b,
 31  *  c, d,
 32  *  tx,ty]
 33  * </pre>
 34  * This is a short form for the 3x3 matrix:
 35  * <pre>
 36  * [a, b, 0
 37  *  c, d, 0
 38  *  tx,ty,1]
 39  * </pre>
 40  * The last column is ignored so the array is shorter and operations are faster.
 41  */
 42 var mat2d = {};
 43 
 44 var mat2dIdentity = new Float32Array([
 45     1, 0,
 46     0, 1,
 47     0, 0
 48 ]);
 49 
 50 /**
 51  * Creates a new identity mat2d
 52  *
 53  * @returns {mat2d} a new 2x3 matrix
 54  */
 55 mat2d.create = function() {
 56     var out = new GLMAT_ARRAY_TYPE(6);
 57     out[0] = 1;
 58     out[1] = 0;
 59     out[2] = 0;
 60     out[3] = 1;
 61     out[4] = 0;
 62     out[5] = 0;
 63     return out;
 64 };
 65 
 66 /**
 67  * Creates a new mat2d initialized with values from an existing matrix
 68  *
 69  * @param {mat2d} a matrix to clone
 70  * @returns {mat2d} a new 2x3 matrix
 71  */
 72 mat2d.clone = function(a) {
 73     var out = new GLMAT_ARRAY_TYPE(6);
 74     out[0] = a[0];
 75     out[1] = a[1];
 76     out[2] = a[2];
 77     out[3] = a[3];
 78     out[4] = a[4];
 79     out[5] = a[5];
 80     return out;
 81 };
 82 
 83 /**
 84  * Copy the values from one mat2d to another
 85  *
 86  * @param {mat2d} out the receiving matrix
 87  * @param {mat2d} a the source matrix
 88  * @returns {mat2d} out
 89  */
 90 mat2d.copy = function(out, a) {
 91     out[0] = a[0];
 92     out[1] = a[1];
 93     out[2] = a[2];
 94     out[3] = a[3];
 95     out[4] = a[4];
 96     out[5] = a[5];
 97     return out;
 98 };
 99 
100 /**
101  * Set a mat2d to the identity matrix
102  *
103  * @param {mat2d} out the receiving matrix
104  * @returns {mat2d} out
105  */
106 mat2d.identity = function(out) {
107     out[0] = 1;
108     out[1] = 0;
109     out[2] = 0;
110     out[3] = 1;
111     out[4] = 0;
112     out[5] = 0;
113     return out;
114 };
115 
116 /**
117  * Inverts a mat2d
118  *
119  * @param {mat2d} out the receiving matrix
120  * @param {mat2d} a the source matrix
121  * @returns {mat2d} out
122  */
123 mat2d.invert = function(out, a) {
124     var aa = a[0], ab = a[1], ac = a[2], ad = a[3],
125         atx = a[4], aty = a[5];
126 
127     var det = aa * ad - ab * ac;
128     if(!det){
129         return null;
130     }
131     det = 1.0 / det;
132 
133     out[0] = ad * det;
134     out[1] = -ab * det;
135     out[2] = -ac * det;
136     out[3] = aa * det;
137     out[4] = (ac * aty - ad * atx) * det;
138     out[5] = (ab * atx - aa * aty) * det;
139     return out;
140 };
141 
142 /**
143  * Calculates the determinant of a mat2d
144  *
145  * @param {mat2d} a the source matrix
146  * @returns {Number} determinant of a
147  */
148 mat2d.determinant = function (a) {
149     return a[0] * a[3] - a[1] * a[2];
150 };
151 
152 /**
153  * Multiplies two mat2d's
154  *
155  * @param {mat2d} out the receiving matrix
156  * @param {mat2d} a the first operand
157  * @param {mat2d} b the second operand
158  * @returns {mat2d} out
159  */
160 mat2d.multiply = function (out, a, b) {
161     var aa = a[0], ab = a[1], ac = a[2], ad = a[3],
162         atx = a[4], aty = a[5],
163         ba = b[0], bb = b[1], bc = b[2], bd = b[3],
164         btx = b[4], bty = b[5];
165 
166     out[0] = aa*ba + ab*bc;
167     out[1] = aa*bb + ab*bd;
168     out[2] = ac*ba + ad*bc;
169     out[3] = ac*bb + ad*bd;
170     out[4] = ba*atx + bc*aty + btx;
171     out[5] = bb*atx + bd*aty + bty;
172     return out;
173 };
174 
175 /**
176  * Alias for {@link mat2d.multiply}
177  * @function
178  */
179 mat2d.mul = mat2d.multiply;

182 /**
183  * Rotates a mat2d by the given angle
184  *
185  * @param {mat2d} out the receiving matrix
186  * @param {mat2d} a the matrix to rotate
187  * @param {Number} rad the angle to rotate the matrix by
188  * @returns {mat2d} out
189  */
190 mat2d.rotate = function (out, a, rad) {
191     var aa = a[0],
192         ab = a[1],
193         ac = a[2],
194         ad = a[3],
195         atx = a[4],
196         aty = a[5],
197         st = Math.sin(rad),
198         ct = Math.cos(rad);
199 
200     out[0] = aa*ct + ab*st;
201     out[1] = -aa*st + ab*ct;
202     out[2] = ac*ct + ad*st;
203     out[3] = -ac*st + ct*ad;
204     out[4] = ct*atx + st*aty;
205     out[5] = ct*aty - st*atx;
206     return out;
207 };

209 /**
210  * Scales the mat2d by the dimensions in the given vec2
211  *
212  * @param {mat2d} out the receiving matrix
213  * @param {mat2d} a the matrix to translate
214  * @param {mat2d} v the vec2 to scale the matrix by
215  * @returns {mat2d} out
216  **/
217 mat2d.scale = function(out, a, v) {
218     var vx = v[0], vy = v[1];
219     out[0] = a[0] * vx;
220     out[1] = a[1] * vy;
221     out[2] = a[2] * vx;
222     out[3] = a[3] * vy;
223     out[4] = a[4] * vx;
224     out[5] = a[5] * vy;
225     return out;
226 };
227 
228 /**
229  * Translates the mat2d by the dimensions in the given vec2
230  *
231  * @param {mat2d} out the receiving matrix
232  * @param {mat2d} a the matrix to translate
233  * @param {mat2d} v the vec2 to translate the matrix by
234  * @returns {mat2d} out
235  **/
236 mat2d.translate = function(out, a, v) {
237     out[0] = a[0];
238     out[1] = a[1];
239     out[2] = a[2];
240     out[3] = a[3];
241     out[4] = a[4] + v[0];
242     out[5] = a[5] + v[1];
243     return out;
244 };

246 /**
247  * Returns a string representation of a mat2d
248  *
249  * @param {mat2d} a matrix to represent as a string
250  * @returns {String} string representation of the matrix
251  */
252 mat2d.str = function (a) {
253     return 'mat2d(' + a[0] + ', ' + a[1] + ', ' + a[2] + ', ' + 
254                     a[3] + ', ' + a[4] + ', ' + a[5] + ')';
255 };
