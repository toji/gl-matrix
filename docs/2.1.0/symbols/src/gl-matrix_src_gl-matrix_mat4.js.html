 42     var out = new GLMAT_ARRAY_TYPE(16);
 43     out[0] = 1;
 44     out[1] = 0;
 45     out[2] = 0;
 46     out[3] = 0;
 47     out[4] = 0;
 48     out[5] = 1;
 49     out[6] = 0;

 59     return out;

142 /**
143  * Transpose the values of a mat4
144  *
145  * @param {mat4} out the receiving matrix
146  * @param {mat4} a the source matrix
147  * @returns {mat4} out
148  */
149 mat4.transpose = function(out, a) {
150     // If we are transposing ourselves we can skip a few steps but have to cache some values
151     if (out === a) {
152         var a01 = a[1], a02 = a[2], a03 = a[3],
153             a12 = a[6], a13 = a[7],
154             a23 = a[11];
155 
156         out[1] = a[4];
157         out[2] = a[8];
158         out[3] = a[12];
159         out[4] = a01;
160         out[6] = a[9];
161         out[7] = a[13];
162         out[8] = a02;
163         out[9] = a12;
164         out[11] = a[14];
165         out[12] = a03;
166         out[13] = a13;
167         out[14] = a23;
168     } else {
169         out[0] = a[0];
170         out[1] = a[4];
171         out[2] = a[8];
172         out[3] = a[12];
173         out[4] = a[1];
174         out[5] = a[5];
175         out[6] = a[9];
176         out[7] = a[13];
177         out[8] = a[2];
178         out[9] = a[6];
179         out[10] = a[10];
180         out[11] = a[14];
181         out[12] = a[3];
182         out[13] = a[7];
183         out[14] = a[11];
184         out[15] = a[15];
185     }
186     
187     return out;
188 };

190 /**
191  * Inverts a mat4
192  *
193  * @param {mat4} out the receiving matrix
194  * @param {mat4} a the source matrix
195  * @returns {mat4} out
196  */
197 mat4.invert = function(out, a) {
198     var a00 = a[0], a01 = a[1], a02 = a[2], a03 = a[3],
199         a10 = a[4], a11 = a[5], a12 = a[6], a13 = a[7],
200         a20 = a[8], a21 = a[9], a22 = a[10], a23 = a[11],
201         a30 = a[12], a31 = a[13], a32 = a[14], a33 = a[15],
202 
203         b00 = a00 * a11 - a01 * a10,
204         b01 = a00 * a12 - a02 * a10,
205         b02 = a00 * a13 - a03 * a10,
206         b03 = a01 * a12 - a02 * a11,
207         b04 = a01 * a13 - a03 * a11,
208         b05 = a02 * a13 - a03 * a12,
209         b06 = a20 * a31 - a21 * a30,
210         b07 = a20 * a32 - a22 * a30,
211         b08 = a20 * a33 - a23 * a30,
212         b09 = a21 * a32 - a22 * a31,
213         b10 = a21 * a33 - a23 * a31,
214         b11 = a22 * a33 - a23 * a32,
215 
216         // Calculate the determinant
217         det = b00 * b11 - b01 * b10 + b02 * b09 + b03 * b08 - b04 * b07 + b05 * b06;
218 
219     if (!det) { 
220         return null; 
221     }
222     det = 1.0 / det;
223 
224     out[0] = (a11 * b11 - a12 * b10 + a13 * b09) * det;
225     out[1] = (a02 * b10 - a01 * b11 - a03 * b09) * det;
226     out[2] = (a31 * b05 - a32 * b04 + a33 * b03) * det;
227     out[3] = (a22 * b04 - a21 * b05 - a23 * b03) * det;
228     out[4] = (a12 * b08 - a10 * b11 - a13 * b07) * det;
229     out[5] = (a00 * b11 - a02 * b08 + a03 * b07) * det;
230     out[6] = (a32 * b02 - a30 * b05 - a33 * b01) * det;
231     out[7] = (a20 * b05 - a22 * b02 + a23 * b01) * det;
232     out[8] = (a10 * b10 - a11 * b08 + a13 * b06) * det;
233     out[9] = (a01 * b08 - a00 * b10 - a03 * b06) * det;
234     out[10] = (a30 * b04 - a31 * b02 + a33 * b00) * det;
235     out[11] = (a21 * b02 - a20 * b04 - a23 * b00) * det;
236     out[12] = (a11 * b07 - a10 * b09 - a12 * b06) * det;
237     out[13] = (a00 * b09 - a01 * b07 + a02 * b06) * det;
238     out[14] = (a31 * b01 - a30 * b03 - a32 * b00) * det;
239     out[15] = (a20 * b03 - a21 * b01 + a22 * b00) * det;
240 
241     return out;
242 };

244 /**
245  * Calculates the adjugate of a mat4
246  *
247  * @param {mat4} out the receiving matrix
248  * @param {mat4} a the source matrix
249  * @returns {mat4} out
250  */
251 mat4.adjoint = function(out, a) {
252     var a00 = a[0], a01 = a[1], a02 = a[2], a03 = a[3],
253         a10 = a[4], a11 = a[5], a12 = a[6], a13 = a[7],
254         a20 = a[8], a21 = a[9], a22 = a[10], a23 = a[11],
255         a30 = a[12], a31 = a[13], a32 = a[14], a33 = a[15];
256 
257     out[0]  =  (a11 * (a22 * a33 - a23 * a32) - a21 * (a12 * a33 - a13 * a32) + a31 * (a12 * a23 - a13 * a22));
258     out[1]  = -(a01 * (a22 * a33 - a23 * a32) - a21 * (a02 * a33 - a03 * a32) + a31 * (a02 * a23 - a03 * a22));
259     out[2]  =  (a01 * (a12 * a33 - a13 * a32) - a11 * (a02 * a33 - a03 * a32) + a31 * (a02 * a13 - a03 * a12));
260     out[3]  = -(a01 * (a12 * a23 - a13 * a22) - a11 * (a02 * a23 - a03 * a22) + a21 * (a02 * a13 - a03 * a12));
261     out[4]  = -(a10 * (a22 * a33 - a23 * a32) - a20 * (a12 * a33 - a13 * a32) + a30 * (a12 * a23 - a13 * a22));
262     out[5]  =  (a00 * (a22 * a33 - a23 * a32) - a20 * (a02 * a33 - a03 * a32) + a30 * (a02 * a23 - a03 * a22));
263     out[6]  = -(a00 * (a12 * a33 - a13 * a32) - a10 * (a02 * a33 - a03 * a32) + a30 * (a02 * a13 - a03 * a12));
264     out[7]  =  (a00 * (a12 * a23 - a13 * a22) - a10 * (a02 * a23 - a03 * a22) + a20 * (a02 * a13 - a03 * a12));
265     out[8]  =  (a10 * (a21 * a33 - a23 * a31) - a20 * (a11 * a33 - a13 * a31) + a30 * (a11 * a23 - a13 * a21));
266     out[9]  = -(a00 * (a21 * a33 - a23 * a31) - a20 * (a01 * a33 - a03 * a31) + a30 * (a01 * a23 - a03 * a21));
267     out[10] =  (a00 * (a11 * a33 - a13 * a31) - a10 * (a01 * a33 - a03 * a31) + a30 * (a01 * a13 - a03 * a11));
268     out[11] = -(a00 * (a11 * a23 - a13 * a21) - a10 * (a01 * a23 - a03 * a21) + a20 * (a01 * a13 - a03 * a11));
269     out[12] = -(a10 * (a21 * a32 - a22 * a31) - a20 * (a11 * a32 - a12 * a31) + a30 * (a11 * a22 - a12 * a21));
270     out[13] =  (a00 * (a21 * a32 - a22 * a31) - a20 * (a01 * a32 - a02 * a31) + a30 * (a01 * a22 - a02 * a21));
271     out[14] = -(a00 * (a11 * a32 - a12 * a31) - a10 * (a01 * a32 - a02 * a31) + a30 * (a01 * a12 - a02 * a11));
272     out[15] =  (a00 * (a11 * a22 - a12 * a21) - a10 * (a01 * a22 - a02 * a21) + a20 * (a01 * a12 - a02 * a11));
273     return out;
274 };

276 /**
277  * Calculates the determinant of a mat4
278  *
279  * @param {mat4} a the source matrix
280  * @returns {Number} determinant of a
281  */
282 mat4.determinant = function (a) {
283     var a00 = a[0], a01 = a[1], a02 = a[2], a03 = a[3],
284         a10 = a[4], a11 = a[5], a12 = a[6], a13 = a[7],
285         a20 = a[8], a21 = a[9], a22 = a[10], a23 = a[11],
286         a30 = a[12], a31 = a[13], a32 = a[14], a33 = a[15],
287 
288         b00 = a00 * a11 - a01 * a10,
289         b01 = a00 * a12 - a02 * a10,
290         b02 = a00 * a13 - a03 * a10,
291         b03 = a01 * a12 - a02 * a11,
292         b04 = a01 * a13 - a03 * a11,
293         b05 = a02 * a13 - a03 * a12,
294         b06 = a20 * a31 - a21 * a30,
295         b07 = a20 * a32 - a22 * a30,
296         b08 = a20 * a33 - a23 * a30,
297         b09 = a21 * a32 - a22 * a31,
298         b10 = a21 * a33 - a23 * a31,
299         b11 = a22 * a33 - a23 * a32;
300 
301     // Calculate the determinant
302     return b00 * b11 - b01 * b10 + b02 * b09 + b03 * b08 - b04 * b07 + b05 * b06;
303 };

305 /**
306  * Multiplies two mat4's
307  *
308  * @param {mat4} out the receiving matrix
309  * @param {mat4} a the first operand
310  * @param {mat4} b the second operand
311  * @returns {mat4} out
312  */
313 mat4.multiply = function (out, a, b) {
314     var a00 = a[0], a01 = a[1], a02 = a[2], a03 = a[3],
315         a10 = a[4], a11 = a[5], a12 = a[6], a13 = a[7],
316         a20 = a[8], a21 = a[9], a22 = a[10], a23 = a[11],
317         a30 = a[12], a31 = a[13], a32 = a[14], a33 = a[15];
318 
319     // Cache only the current line of the second matrix
320     var b0  = b[0], b1 = b[1], b2 = b[2], b3 = b[3];  
321     out[0] = b0*a00 + b1*a10 + b2*a20 + b3*a30;
322     out[1] = b0*a01 + b1*a11 + b2*a21 + b3*a31;
323     out[2] = b0*a02 + b1*a12 + b2*a22 + b3*a32;
324     out[3] = b0*a03 + b1*a13 + b2*a23 + b3*a33;
325 
326     b0 = b[4]; b1 = b[5]; b2 = b[6]; b3 = b[7];
327     out[4] = b0*a00 + b1*a10 + b2*a20 + b3*a30;
328     out[5] = b0*a01 + b1*a11 + b2*a21 + b3*a31;
329     out[6] = b0*a02 + b1*a12 + b2*a22 + b3*a32;
330     out[7] = b0*a03 + b1*a13 + b2*a23 + b3*a33;
331 
332     b0 = b[8]; b1 = b[9]; b2 = b[10]; b3 = b[11];
333     out[8] = b0*a00 + b1*a10 + b2*a20 + b3*a30;
334     out[9] = b0*a01 + b1*a11 + b2*a21 + b3*a31;
335     out[10] = b0*a02 + b1*a12 + b2*a22 + b3*a32;
336     out[11] = b0*a03 + b1*a13 + b2*a23 + b3*a33;
337 
338     b0 = b[12]; b1 = b[13]; b2 = b[14]; b3 = b[15];
339     out[12] = b0*a00 + b1*a10 + b2*a20 + b3*a30;
340     out[13] = b0*a01 + b1*a11 + b2*a21 + b3*a31;
341     out[14] = b0*a02 + b1*a12 + b2*a22 + b3*a32;
342     out[15] = b0*a03 + b1*a13 + b2*a23 + b3*a33;
343     return out;
344 };

352 /**
353  * Translate a mat4 by the given vector
354  *
355  * @param {mat4} out the receiving matrix
356  * @param {mat4} a the matrix to translate
357  * @param {vec3} v vector to translate by
358  * @returns {mat4} out
359  */
360 mat4.translate = function (out, a, v) {
361     var x = v[0], y = v[1], z = v[2],
362         a00, a01, a02, a03,
363         a10, a11, a12, a13,
364         a20, a21, a22, a23;
365 
366     if (a === out) {
367         out[12] = a[0] * x + a[4] * y + a[8] * z + a[12];
368         out[13] = a[1] * x + a[5] * y + a[9] * z + a[13];
369         out[14] = a[2] * x + a[6] * y + a[10] * z + a[14];
370         out[15] = a[3] * x + a[7] * y + a[11] * z + a[15];
371     } else {
372         a00 = a[0]; a01 = a[1]; a02 = a[2]; a03 = a[3];
373         a10 = a[4]; a11 = a[5]; a12 = a[6]; a13 = a[7];
374         a20 = a[8]; a21 = a[9]; a22 = a[10]; a23 = a[11];
375 
376         out[0] = a00; out[1] = a01; out[2] = a02; out[3] = a03;
377         out[4] = a10; out[5] = a11; out[6] = a12; out[7] = a13;
378         out[8] = a20; out[9] = a21; out[10] = a22; out[11] = a23;
379 
380         out[12] = a00 * x + a10 * y + a20 * z + a[12];
381         out[13] = a01 * x + a11 * y + a21 * z + a[13];
382         out[14] = a02 * x + a12 * y + a22 * z + a[14];
383         out[15] = a03 * x + a13 * y + a23 * z + a[15];
384     }
385 
386     return out;
387 };

389 /**
390  * Scales the mat4 by the dimensions in the given vec3

395  * @returns {mat4} out
396  **/
397 mat4.scale = function(out, a, v) {
398     var x = v[0], y = v[1], z = v[2];
399 
400     out[0] = a[0] * x;

402     out[2] = a[2] * x;
403     out[3] = a[3] * x;

405     out[5] = a[5] * y;
406     out[6] = a[6] * y;
407     out[7] = a[7] * y;
408     out[8] = a[8] * z;
409     out[9] = a[9] * z;
410     out[10] = a[10] * z;
411     out[11] = a[11] * z;
412     out[12] = a[12];
413     out[13] = a[13];
414     out[14] = a[14];
415     out[15] = a[15];
416     return out;
417 };

419 /**
420  * Rotates a mat4 by the given angle
421  *
422  * @param {mat4} out the receiving matrix
423  * @param {mat4} a the matrix to rotate
424  * @param {Number} rad the angle to rotate the matrix by
425  * @param {vec3} axis the axis to rotate around
426  * @returns {mat4} out
427  */
428 mat4.rotate = function (out, a, rad, axis) {
429     var x = axis[0], y = axis[1], z = axis[2],
430         len = Math.sqrt(x * x + y * y + z * z),
431         s, c, t,
432         a00, a01, a02, a03,
433         a10, a11, a12, a13,
434         a20, a21, a22, a23,
435         b00, b01, b02,
436         b10, b11, b12,
437         b20, b21, b22;
438 
439     if (Math.abs(len) < GLMAT_EPSILON) { return null; }
440     
441     len = 1 / len;
442     x *= len;
443     y *= len;
444     z *= len;
445 
446     s = Math.sin(rad);
447     c = Math.cos(rad);
448     t = 1 - c;
449 
450     a00 = a[0]; a01 = a[1]; a02 = a[2]; a03 = a[3];
451     a10 = a[4]; a11 = a[5]; a12 = a[6]; a13 = a[7];
452     a20 = a[8]; a21 = a[9]; a22 = a[10]; a23 = a[11];
453 
454     // Construct the elements of the rotation matrix
455     b00 = x * x * t + c; b01 = y * x * t + z * s; b02 = z * x * t - y * s;
456     b10 = x * y * t - z * s; b11 = y * y * t + c; b12 = z * y * t + x * s;
457     b20 = x * z * t + y * s; b21 = y * z * t - x * s; b22 = z * z * t + c;
458 
459     // Perform rotation-specific matrix multiplication
460     out[0] = a00 * b00 + a10 * b01 + a20 * b02;
461     out[1] = a01 * b00 + a11 * b01 + a21 * b02;
462     out[2] = a02 * b00 + a12 * b01 + a22 * b02;
463     out[3] = a03 * b00 + a13 * b01 + a23 * b02;
464     out[4] = a00 * b10 + a10 * b11 + a20 * b12;
465     out[5] = a01 * b10 + a11 * b11 + a21 * b12;
466     out[6] = a02 * b10 + a12 * b11 + a22 * b12;
467     out[7] = a03 * b10 + a13 * b11 + a23 * b12;
468     out[8] = a00 * b20 + a10 * b21 + a20 * b22;
469     out[9] = a01 * b20 + a11 * b21 + a21 * b22;
470     out[10] = a02 * b20 + a12 * b21 + a22 * b22;
471     out[11] = a03 * b20 + a13 * b21 + a23 * b22;
472 
473     if (a !== out) { // If the source and destination differ, copy the unchanged last row
474         out[12] = a[12];
475         out[13] = a[13];
476         out[14] = a[14];
477         out[15] = a[15];
478     }
479     return out;
480 };

482 /**
483  * Rotates a matrix by the given angle around the X axis
484  *
485  * @param {mat4} out the receiving matrix
486  * @param {mat4} a the matrix to rotate
487  * @param {Number} rad the angle to rotate the matrix by
488  * @returns {mat4} out
489  */
490 mat4.rotateX = function (out, a, rad) {
491     var s = Math.sin(rad),
492         c = Math.cos(rad),
493         a10 = a[4],
494         a11 = a[5],
495         a12 = a[6],
496         a13 = a[7],
497         a20 = a[8],
498         a21 = a[9],
499         a22 = a[10],
500         a23 = a[11];
501 
502     if (a !== out) { // If the source and destination differ, copy the unchanged rows
503         out[0]  = a[0];
504         out[1]  = a[1];
505         out[2]  = a[2];
506         out[3]  = a[3];
507         out[12] = a[12];
508         out[13] = a[13];
509         out[14] = a[14];
510         out[15] = a[15];
511     }
512 
513     // Perform axis-specific matrix multiplication
514     out[4] = a10 * c + a20 * s;
515     out[5] = a11 * c + a21 * s;
516     out[6] = a12 * c + a22 * s;
517     out[7] = a13 * c + a23 * s;
518     out[8] = a20 * c - a10 * s;
519     out[9] = a21 * c - a11 * s;
520     out[10] = a22 * c - a12 * s;
521     out[11] = a23 * c - a13 * s;
522     return out;
523 };

526  * Rotates a matrix by the given angle around the Y axis
527  *
528  * @param {mat4} out the receiving matrix
529  * @param {mat4} a the matrix to rotate
530  * @param {Number} rad the angle to rotate the matrix by
531  * @returns {mat4} out
532  */
533 mat4.rotateY = function (out, a, rad) {
534     var s = Math.sin(rad),
535         c = Math.cos(rad),
536         a00 = a[0],
537         a01 = a[1],
538         a02 = a[2],
539         a03 = a[3],
540         a20 = a[8],
541         a21 = a[9],
542         a22 = a[10],
543         a23 = a[11];
544 
545     if (a !== out) { // If the source and destination differ, copy the unchanged rows
546         out[4]  = a[4];
547         out[5]  = a[5];
548         out[6]  = a[6];
549         out[7]  = a[7];
550         out[12] = a[12];
551         out[13] = a[13];
552         out[14] = a[14];
553         out[15] = a[15];
554     }
555 
556     // Perform axis-specific matrix multiplication
557     out[0] = a00 * c - a20 * s;
558     out[1] = a01 * c - a21 * s;
559     out[2] = a02 * c - a22 * s;
560     out[3] = a03 * c - a23 * s;
561     out[8] = a00 * s + a20 * c;
562     out[9] = a01 * s + a21 * c;
563     out[10] = a02 * s + a22 * c;
564     out[11] = a03 * s + a23 * c;
565     return out;
566 };

611 /**
612  * Creates a matrix from a quaternion rotation and vector translation
613  * This is equivalent to (but much faster than):
614  *
615  *     mat4.identity(dest);
616  *     mat4.translate(dest, vec);
617  *     var quatMat = mat4.create();
618  *     quat4.toMat4(quat, quatMat);
619  *     mat4.multiply(dest, quatMat);
620  *
621  * @param {mat4} out mat4 receiving operation result
622  * @param {quat4} q Rotation quaternion
623  * @param {vec3} v Translation vector
624  * @returns {mat4} out
625  */
626 mat4.fromRotationTranslation = function (out, q, v) {
627     // Quaternion math
628     var x = q[0], y = q[1], z = q[2], w = q[3],
629         x2 = x + x,
630         y2 = y + y,
631         z2 = z + z,
632 
633         xx = x * x2,
634         xy = x * y2,
635         xz = x * z2,
636         yy = y * y2,
637         yz = y * z2,
638         zz = z * z2,
639         wx = w * x2,
640         wy = w * y2,
641         wz = w * z2;
642 
643     out[0] = 1 - (yy + zz);
644     out[1] = xy + wz;
645     out[2] = xz - wy;
646     out[3] = 0;
647     out[4] = xy - wz;
648     out[5] = 1 - (xx + zz);
649     out[6] = yz + wx;
650     out[7] = 0;
651     out[8] = xz + wy;
652     out[9] = yz - wx;
653     out[10] = 1 - (xx + yy);
654     out[11] = 0;
655     out[12] = v[0];
656     out[13] = v[1];
657     out[14] = v[2];
658     out[15] = 1;
659     
660     return out;
661 };

664 * Calculates a 4x4 matrix from the given quaternion
665 *
666 * @param {mat4} out mat4 receiving operation result
667 * @param {quat} q Quaternion to create matrix from
668 *
669 * @returns {mat4} out
670 */
671 mat4.fromQuat = function (out, q) {

673         x2 = x + x,
674         y2 = y + y,
675         z2 = z + z,
676 
677         xx = x * x2,
678         xy = x * y2,
679         xz = x * z2,
680         yy = y * y2,
681         yz = y * z2,
682         zz = z * z2,
683         wx = w * x2,
684         wy = w * y2,
685         wz = w * z2;
686 
687     out[0] = 1 - (yy + zz);
688     out[1] = xy + wz;
689     out[2] = xz - wy;
690     out[3] = 0;

692     out[4] = xy - wz;
693     out[5] = 1 - (xx + zz);
694     out[6] = yz + wx;
695     out[7] = 0;

698     out[9] = yz - wx;
699     out[10] = 1 - (xx + yy);

703     out[13] = 0;
704     out[14] = 0;

710 /**
711  * Generates a frustum matrix with the given bounds
712  *
713  * @param {mat4} out mat4 frustum matrix will be written into
714  * @param {Number} left Left bound of the frustum
715  * @param {Number} right Right bound of the frustum
716  * @param {Number} bottom Bottom bound of the frustum
717  * @param {Number} top Top bound of the frustum
718  * @param {Number} near Near bound of the frustum
719  * @param {Number} far Far bound of the frustum
720  * @returns {mat4} out
721  */
722 mat4.frustum = function (out, left, right, bottom, top, near, far) {
723     var rl = 1 / (right - left),
724         tb = 1 / (top - bottom),
725         nf = 1 / (near - far);
726     out[0] = (near * 2) * rl;
727     out[1] = 0;
728     out[2] = 0;
729     out[3] = 0;
730     out[4] = 0;
731     out[5] = (near * 2) * tb;
732     out[6] = 0;
733     out[7] = 0;
734     out[8] = (right + left) * rl;
735     out[9] = (top + bottom) * tb;
736     out[10] = (far + near) * nf;
737     out[11] = -1;
738     out[12] = 0;
739     out[13] = 0;
740     out[14] = (far * near * 2) * nf;
741     out[15] = 0;
742     return out;
743 };

745 /**
746  * Generates a perspective projection matrix with the given bounds
747  *
748  * @param {mat4} out mat4 frustum matrix will be written into
749  * @param {number} fovy Vertical field of view in radians
750  * @param {number} aspect Aspect ratio. typically viewport width/height
751  * @param {number} near Near bound of the frustum
752  * @param {number} far Far bound of the frustum
753  * @returns {mat4} out
754  */
755 mat4.perspective = function (out, fovy, aspect, near, far) {
756     var f = 1.0 / Math.tan(fovy / 2),
757         nf = 1 / (near - far);
758     out[0] = f / aspect;
759     out[1] = 0;
760     out[2] = 0;
761     out[3] = 0;
762     out[4] = 0;
763     out[5] = f;
764     out[6] = 0;
765     out[7] = 0;
766     out[8] = 0;
767     out[9] = 0;
768     out[10] = (far + near) * nf;
769     out[11] = -1;
770     out[12] = 0;
771     out[13] = 0;
772     out[14] = (2 * far * near) * nf;
773     out[15] = 0;
774     return out;
775 };

777 /**
778  * Generates a orthogonal projection matrix with the given bounds
779  *
780  * @param {mat4} out mat4 frustum matrix will be written into
781  * @param {number} left Left bound of the frustum
782  * @param {number} right Right bound of the frustum
783  * @param {number} bottom Bottom bound of the frustum
784  * @param {number} top Top bound of the frustum
785  * @param {number} near Near bound of the frustum
786  * @param {number} far Far bound of the frustum
787  * @returns {mat4} out
788  */
789 mat4.ortho = function (out, left, right, bottom, top, near, far) {
790     var lr = 1 / (left - right),
791         bt = 1 / (bottom - top),
792         nf = 1 / (near - far);
793     out[0] = -2 * lr;
794     out[1] = 0;
795     out[2] = 0;
796     out[3] = 0;
797     out[4] = 0;
798     out[5] = -2 * bt;
799     out[6] = 0;
800     out[7] = 0;
801     out[8] = 0;
802     out[9] = 0;
803     out[10] = 2 * nf;
804     out[11] = 0;
805     out[12] = (left + right) * lr;
806     out[13] = (top + bottom) * bt;
807     out[14] = (far + near) * nf;
808     out[15] = 1;
809     return out;
810 };

812 /**
813  * Generates a look-at matrix with the given eye position, focal point, and up axis
814  *
815  * @param {mat4} out mat4 frustum matrix will be written into
816  * @param {vec3} eye Position of the viewer
817  * @param {vec3} center Point the viewer is looking at
818  * @param {vec3} up vec3 pointing up
819  * @returns {mat4} out
820  */
821 mat4.lookAt = function (out, eye, center, up) {
822     var x0, x1, x2, y0, y1, y2, z0, z1, z2, len,
823         eyex = eye[0],
824         eyey = eye[1],
825         eyez = eye[2],
826         upx = up[0],
827         upy = up[1],
828         upz = up[2],
829         centerx = center[0],
830         centery = center[1],
831         centerz = center[2];
832 
833     if (Math.abs(eyex - centerx) < GLMAT_EPSILON &&
834         Math.abs(eyey - centery) < GLMAT_EPSILON &&
835         Math.abs(eyez - centerz) < GLMAT_EPSILON) {
836         return mat4.identity(out);
837     }
838 
839     z0 = eyex - centerx;
840     z1 = eyey - centery;
841     z2 = eyez - centerz;
842 
843     len = 1 / Math.sqrt(z0 * z0 + z1 * z1 + z2 * z2);
844     z0 *= len;
845     z1 *= len;
846     z2 *= len;
847 
848     x0 = upy * z2 - upz * z1;
849     x1 = upz * z0 - upx * z2;
850     x2 = upx * z1 - upy * z0;
851     len = Math.sqrt(x0 * x0 + x1 * x1 + x2 * x2);
852     if (!len) {
853         x0 = 0;
854         x1 = 0;
855         x2 = 0;
856     } else {
857         len = 1 / len;
858         x0 *= len;
859         x1 *= len;
860         x2 *= len;
861     }
862 
863     y0 = z1 * x2 - z2 * x1;
864     y1 = z2 * x0 - z0 * x2;
865     y2 = z0 * x1 - z1 * x0;
866 
867     len = Math.sqrt(y0 * y0 + y1 * y1 + y2 * y2);
868     if (!len) {
869         y0 = 0;
870         y1 = 0;
871         y2 = 0;
872     } else {
873         len = 1 / len;
874         y0 *= len;
875         y1 *= len;
876         y2 *= len;
877     }
878 
879     out[0] = x0;
880     out[1] = y0;
881     out[2] = z0;
882     out[3] = 0;
883     out[4] = x1;
884     out[5] = y1;
885     out[6] = z1;
886     out[7] = 0;
887     out[8] = x2;
888     out[9] = y2;
889     out[10] = z2;
890     out[11] = 0;
891     out[12] = -(x0 * eyex + x1 * eyey + x2 * eyez);
892     out[13] = -(y0 * eyex + y1 * eyey + y2 * eyez);
893     out[14] = -(z0 * eyex + z1 * eyey + z2 * eyez);
894     out[15] = 1;
895 
896     return out;
897 };
